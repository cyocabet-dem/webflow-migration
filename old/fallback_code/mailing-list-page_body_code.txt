<script>
// ============================================
// MAILING LIST FORM CONTROLLER
// ============================================

(function() {
  'use strict';

  // Locale helper (prefers shared helper from site-wide-footer.js)
  function isNL() {
    if (window.DematI18n && window.DematI18n.isNL) return window.DematI18n.isNL();
    return (document.documentElement.lang || '').toLowerCase().indexOf('nl') === 0;
  }

  // Strings the controller injects at runtime
  const T = {
    subscribe:     { en: 'Subscribe',      nl: 'Inschrijven' },
    subscribing:   { en: 'Subscribing...', nl: 'Inschrijven...' },
    firstNameReq:  { en: 'Please enter your first name.', nl: 'Vul je voornaam in.' },
    lastNameReq:   { en: 'Please enter your last name.',  nl: 'Vul je achternaam in.' },
    emailReq:      { en: 'Please enter your email address.', nl: 'Vul je e-mailadres in.' },
    emailInvalid:  { en: 'Please enter a valid email address.', nl: 'Vul een geldig e-mailadres in.' },
    termsReq:      { en: 'Please agree to the Terms & Conditions and Privacy Policy.', nl: 'Ga akkoord met de algemene voorwaarden en het privacybeleid.' },
    already:       { en: 'This email is already subscribed to our mailing list.', nl: 'Dit e-mailadres is al ingeschreven voor onze mailinglijst.' },
    generic:       { en: 'Something went wrong. Please try again.', nl: 'Er ging iets mis. Probeer het opnieuw.' },
    phFirstName:   { en: 'First Name',     nl: 'Voornaam' },
    phLastName:    { en: 'Last Name',      nl: 'Achternaam' }
    // Note: email placeholder 'your@email.com' is language-neutral, left as-is
  };
  function t(key) {
    const e = T[key];
    return e ? (isNL() ? e.nl : e.en) : '';
  }

  // DOM Elements
  const form = document.getElementById('mailingListForm');
  const submitBtn = document.getElementById('mailingSubmit');
  const errorEl = document.getElementById('mailingError');
  const successEl = document.getElementById('mailingSuccess');
  const firstNameInput = document.getElementById('firstName');
  const lastNameInput = document.getElementById('lastName');
  const emailInput = document.getElementById('emailAddress');
  const termsCheckbox = document.getElementById('termsCheckbox');

  if (!form) {
    console.warn('Mailing list form not found');
    return;
  }

  // Localize placeholders on the Dutch site (can't put spans on attributes)
  if (isNL()) {
    if (firstNameInput) firstNameInput.placeholder = t('phFirstName');
    if (lastNameInput) lastNameInput.placeholder = t('phLastName');
  }

  // Show error message
  function showError(message) {
    errorEl.textContent = message;
    errorEl.classList.add('visible');
  }

  // Hide error message
  function hideError() {
    errorEl.classList.remove('visible');
  }

  // Show success state
  function showSuccess() {
    form.classList.add('hidden');
    successEl.classList.add('visible');
  }

  // Validate email format
  function isValidEmail(email) {
    return /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email);
  }

  // Clear error styling from inputs
  function clearInputErrors() {
    [firstNameInput, lastNameInput, emailInput].forEach(input => {
      input.classList.remove('error');
    });
  }

  // Validate form
  function validateForm() {
    hideError();
    clearInputErrors();

    // Check first name
    if (!firstNameInput.value.trim()) {
      firstNameInput.classList.add('error');
      showError(t('firstNameReq'));
      firstNameInput.focus();
      return false;
    }

    // Check last name
    if (!lastNameInput.value.trim()) {
      lastNameInput.classList.add('error');
      showError(t('lastNameReq'));
      lastNameInput.focus();
      return false;
    }

    // Check email
    if (!emailInput.value.trim()) {
      emailInput.classList.add('error');
      showError(t('emailReq'));
      emailInput.focus();
      return false;
    }

    if (!isValidEmail(emailInput.value.trim())) {
      emailInput.classList.add('error');
      showError(t('emailInvalid'));
      emailInput.focus();
      return false;
    }

    // Check terms checkbox
    if (!termsCheckbox.checked) {
      showError(t('termsReq'));
      return false;
    }

    return true;
  }

  // Handle form submission
  async function handleSubmit(event) {
    event.preventDefault();

    if (!validateForm()) {
      return;
    }

    // Disable button and show loading state
    submitBtn.disabled = true;
    submitBtn.textContent = t('subscribing');
    hideError();

    // Prepare data
    const formData = {
      first_name: firstNameInput.value.trim(),
      last_name: lastNameInput.value.trim(),
      email: emailInput.value.trim()
    };

    try {
      const response = await fetch(`${window.API_BASE_URL}/mailing-list/subscribe`, {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json'
        },
        body: JSON.stringify(formData)
      });

      const data = await response.json().catch(() => ({}));

      if (!response.ok) {
        // Handle specific error messages
        if (response.status === 409 || (data.message && data.message.toLowerCase().includes('already'))) {
          throw new Error(t('already'));
        } else if (data.message) {
          throw new Error(data.message);
        } else {
          throw new Error(t('generic'));
        }
      }

      // Success!
      console.log('✅ Successfully subscribed:', formData.email);
      showSuccess();

    } catch (error) {
      console.error('Mailing list signup error:', error);
      showError(error.message || t('generic'));
      
      // Re-enable button
      submitBtn.disabled = false;
      submitBtn.textContent = t('subscribe');
    }
  }

  // Clear error on input
  function handleInputChange(event) {
    event.target.classList.remove('error');
    hideError();
  }

  // Event listeners
  form.addEventListener('submit', handleSubmit);

  [firstNameInput, lastNameInput, emailInput].forEach(input => {
    input.addEventListener('input', handleInputChange);
  });

  termsCheckbox.addEventListener('change', hideError);

  console.log('📧 Mailing list form initialized');

})();
</script>
